\section{Regime Refinement}
\label{sec:refinement}

This section derives required refinements of regime profiles
under transformation-driven identity constraints.
Refinement occurs when no single transformation-classification assignment
is consistent across all admissible cases within a provisional regime.

%--------------------------------------------
\subsection{Refinement Principle}

\begin{definition}[Refinement condition]
\label{def:refinement_condition}
A regime profile $P$ requires refinement if there exists a transformation family
$f \in \mathbb{F}$ such that no single classification
\[
\mathrm{Class}_P(f) \in \{ \mathrm{IGN}, \mathrm{PRS}, \mathrm{BRK} \}
\]
is consistent with all admissible instances assigned to $P$.
\end{definition}

\begin{definition}[Forced refinement]
\label{def:forced_refinement}
A refinement is forced if failure to refine yields violation
of stable reference.
\end{definition}


Refinement separates a provisional regime into distinct profiles
that realize incompatible transformation-classification patterns.


%--------------------------------------------
\subsection{ENR Refinement}

\begin{proposition}[ENR split under branching]
\label{prop:enr_split}
The regime of enduring non-normative referents requires refinement
under branching transformations $\mathrm{BF}$.
\end{proposition}

\begin{proof}
Under $\mathrm{BF}$, an identity carrier may produce multiple continuations.
In some admissible cases, identity must be preserved across branches;
in others, branching produces distinct identities.

No single classification of $\mathrm{BF}$ as $\mathrm{PRS}$ or $\mathrm{BRK}$
is consistent across all such cases.
Therefore, refinement is required.
\end{proof}

\begin{definition}[ENR refinement]
\label{def:enr_refinement}
The ENR regime splits into:
\begin{itemize}
\item $\mathrm{ENR\mbox{-}L}$: identity bound to locus (branching breaks identity),
\item $\mathrm{ENR\mbox{-}I}$: identity bound to instrument (branching preserves identity across instances).
\end{itemize}
\end{definition}

%--------------------------------------------
\subsection{CTX Refinement}

\begin{proposition}[CTX split under decomposition]
\label{prop:ctx_split}
The regime of applicability contexts requires refinement
under aggregation/decomposition transformations $\mathrm{AD}$.
\end{proposition}

\begin{proof}
Under $\mathrm{AD}$, a context may be decomposed into subcontexts.
In some admissible cases, identity is preserved under decomposition;
in others, decomposition alters applicability conditions.

No single classification of $\mathrm{AD}$ as $\mathrm{PRS}$ or $\mathrm{BRK}$
is consistent across all admissible cases.
Therefore, refinement is required.
\end{proof}

\begin{definition}[CTX refinement]
\label{def:ctx_refinement}
The CTX regime splits into:
\begin{itemize}
\item $\mathrm{CTX\mbox{-}E}$: extension-sensitive contexts,
\item $\mathrm{CTX\mbox{-}S}$: structure-sensitive contexts.
\end{itemize}
\end{definition}

%--------------------------------------------
\subsection{NOR Refinement}

\begin{proposition}[NOR split under refinement]
\label{prop:nor_split}
The regime of normative structures requires refinement
under refinement transformations $\mathrm{RF}$.
\end{proposition}

\begin{proof}
Under $\mathrm{RF}$, a structure may be refined into more detailed form.
In some admissible cases, identity is preserved under refinement;
in others, refinement alters structural composition in a way
that changes identity.

No single classification of $\mathrm{RF}$ as $\mathrm{PRS}$ or $\mathrm{BRK}$
is consistent across all admissible cases.
Therefore, refinement is required.
\end{proof}

\begin{definition}[NOR refinement]
\label{def:nor_refinement}
The NOR regime splits into:
\begin{itemize}
\item $\mathrm{NOR\mbox{-}C}$: composition-sensitive structures,
\item $\mathrm{NOR\mbox{-}S}$: substitution-sensitive structures.
\end{itemize}
\end{definition}

%--------------------------------------------
\subsection{Summary of Refinement}

\begin{theorem}[Refinement result]
\label{thm:refinement_result}
The provisional regime set requires refinement into distinct profiles
for ENR, CTX, and NOR under transformation families
$\mathrm{BF}$, $\mathrm{AD}$, and $\mathrm{RF}$ respectively.
\end{theorem}

\begin{proof}
Propositions~\ref{prop:enr_split}, \ref{prop:ctx_split},
and \ref{prop:nor_split} establish that no single profile
can realize consistent classification for the corresponding
transformation families.
Therefore, refinement into distinct profiles is required.
\end{proof}


Refinement is driven entirely by transformation-classification inconsistency.
No additional ontological distinctions are introduced.
